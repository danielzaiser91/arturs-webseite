# Die eigentliche "nicht indexieren"-Sperre sitzt als <meta name="robots"
# content="noindex, nofollow"> auf jeder Seite -- das reicht aus, um
# Google & Co. von der Indexierung abzuhalten, ohne den Zugriff für
# andere Bots (z. B. Link-Vorschau-Crawler von WhatsApp/Telegram) zu
# blockieren, wie es ein pauschales "Disallow: /" hier tun würde.
User-agent: *
Allow: /
